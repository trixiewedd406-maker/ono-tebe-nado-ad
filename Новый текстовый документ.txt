1. Android: кнопка и заголовок
activity_main.xml
<?xml version="1.0" encoding="utf-8"?>
<LinearLayout
    xmlns:android="http://schemas.android.com/apk/res/android"
    android:layout_width="match_parent"
    android:layout_height="match_parent"
    android:orientation="vertical"
    android:gravity="center">

    <TextView
        android:id="@+id/textView"
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:text="Заголовок"
        android:textSize="24sp"/>

    <Button
        android:id="@+id/button"
        android:layout_width="wrap_content"
        android:layout_height="wrap_content"
        android:text="Нажми"/>

</LinearLayout>
MainActivity.java
package com.example.myapp;

import android.os.Bundle;
import android.widget.Button;
import android.widget.TextView;
import androidx.appcompat.app.AppCompatActivity;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        TextView textView = findViewById(R.id.textView);
        Button button = findViewById(R.id.button);

        button.setOnClickListener(v -> {
            textView.setText("Привет!");
        });
    }
}

7. Android: увеличение размера текста
Java
TextView textView = findViewById(R.id.textView);
Button button = findViewById(R.id.button);

button.setOnClickListener(v -> {
    float size = textView.getTextSize();
    textView.setTextSize(size / getResources().getDisplayMetrics().scaledDensity + 2);
});
8. Android: переключение цвета фона
XML
<LinearLayout
    android:id="@+id/layout"
    ...
>
Java
LinearLayout layout = findViewById(R.id.layout);
Button button = findViewById(R.id.button);

boolean isWhite = true;

button.setOnClickListener(v -> {
    if(isWhite){
        layout.setBackgroundColor(Color.LTGRAY);
    } else {
        layout.setBackgroundColor(Color.WHITE);
    }
    isWhite = !isWhite;
});

Не забудьте импорт:

import android.graphics.Color;

13. Java: функция суммы

Создайте файл Main.java.

public class Main {

    public static int sum(int a, int b){
        return a + b;
    }

    public static void main(String[] args) {
        System.out.println(sum(2, 3));
    }
}

15. Unit-тест JUnit
Класс SumUtil.java
public class SumUtil {

    public static int sum(int a, int b){
        return a + b;
    }
}
Класс SumUtilTest.java

Для JUnit 5:

import org.junit.jupiter.api.Test;
import static org.junit.jupiter.api.Assertions.assertEquals;

public class SumUtilTest {

    @Test
    public void testSum() {

        assertEquals(
                5,
                SumUtil.sum(2, 3)
        );

    }
}
Что проверяет тест
SumUtil.sum(2,3)

возвращает:

5
